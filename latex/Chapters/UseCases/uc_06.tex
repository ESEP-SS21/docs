    \addTextCollum{Verwandte Requirements}{

    }
    \addTextCollum{Description}{

    }
    \addTextCollum{Actors}{
        \gls{sortierer}
    }
    \addCollum{Precondition}{

    }

    \addCollum{Postcondition}{
        \item \gls{sortierer} zurückgesetzt
        \item \gls{workpiece} nicht aussortiert
    }
